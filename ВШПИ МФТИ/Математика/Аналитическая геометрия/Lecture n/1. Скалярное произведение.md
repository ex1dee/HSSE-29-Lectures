$$a_{1} = \frac{(\bar{a}, \bar{e}_{1})}{|e_{1}|^2}, a_{2} = \frac{(\bar{a}, \bar{e}_{2})}{|e_{2}|^2}, ... \implies \bar{a} = \frac{(\bar{a}, \bar{e}_{1})}{|e_{1}|^2}*\bar{e}_{1} + \frac{(\bar{a}, \bar{e}_{2})}{|e_{2}|^2}*\bar{e}_{2} +  \dots$$
